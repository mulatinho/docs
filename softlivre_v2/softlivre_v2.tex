\begin{frame}
	\frametitle{Conceitos Básicos}
	\framesubtitle{Qual a definição para \large{\textbf{software}}!?}
	
	\emph{"Qualquer programa ou grupo de programas que instrui o hardware sobre a 
	maneira como ele deve executar uma tarefa, inclusive sistemas operacionais,
	processadores de texto e programas de aplicação."} Dicionário Michaelis
	
	\vskip 5mm Ou ainda... \vskip 5mm
	
	\emph{"Uma série de instruções definidas por uma linguagem de programação que
	com uma lógica escrita por um ou mais desenvolvedor(es) executa operações em
	sistematicamente afim de se atingir um objetivo específico."}
\end{frame}

\begin{frame}
	\frametitle{Conceitos Básicos}
	\framesubtitle{Principais requisitos}
	
	O que é \textbf{necessário para construir} um software!?
	
	\begin{enumerate}
		\item Um hardware (computador, sistema embarcado, etc.)
		\item Um compilador (transformará o código em binário)
		\item Um desenvolvedor (responsável pelo código do programa)
	\end{enumerate}
	
	Geralmente existe um sistema operacional entre os itens 1 e 2, embora isso
	não seja necessariamente a regra. 
	
	\begin{block}{Arquiteturas de Computadores}
		x86, ia64, SPARC, PowerPC, ARM, Z80, ....
	\end{block}
